param(
    [int]$MaxThreads    = 10,
    [string]$OutputPath = (Split-Path -Parent $MyInvocation.MyCommand.Path),
    [string]$SearchBase = ""
)

$ErrorActionPreference = 'Continue'
$WarningPreference     = 'SilentlyContinue'

# ─────────────────────────────────────────────
# Pre-flight checks
# ─────────────────────────────────────────────
$currentIdentity  = [Security.Principal.WindowsIdentity]::GetCurrent()
$currentPrincipal = [Security.Principal.WindowsPrincipal]$currentIdentity
$adminRole        = [Security.Principal.WindowsBuiltInRole]::Administrator

if (-not $currentPrincipal.IsInRole($adminRole)) {
    Write-Host "ERROR: This script must be run as Administrator." -ForegroundColor Red
    exit 1
}

try {
    Import-Module ActiveDirectory -ErrorAction Stop
}
catch {
    Write-Host "ERROR: Active Directory module not found. Install RSAT tools and retry." -ForegroundColor Red
    exit 1
}

if (-not (Test-Path $OutputPath)) {
    Write-Host "ERROR: Output path does not exist: $OutputPath" -ForegroundColor Red
    exit 1
}

# ─────────────────────────────────────────────
# Configuration
# ─────────────────────────────────────────────
$timestamp = Get-Date -Format "yyyyMMdd_HHmmss"
$csvPath   = Join-Path $OutputPath "AD_Users_$timestamp.csv"

Write-Host ""
Write-Host "╔══════════════════════════════════════════════╗" -ForegroundColor Cyan
Write-Host "║     Active Directory User Export Script      ║" -ForegroundColor Cyan
Write-Host "╚══════════════════════════════════════════════╝" -ForegroundColor Cyan

# Auto-detect domain root
try {
    $domainDN = (Get-ADDomain -ErrorAction Stop).DistinguishedName
    Write-Host "  Domain DN   : $domainDN" -ForegroundColor Gray
}
catch {
    Write-Host "ERROR: Unable to contact Active Directory." -ForegroundColor Red
    Write-Host "  Detail: $_" -ForegroundColor Red
    exit 1
}

# Resolve SearchBase
if (-not [string]::IsNullOrWhiteSpace($SearchBase)) {
    Write-Host "  Search Base : $SearchBase (manual override)" -ForegroundColor Gray
}
else {
    $usersOU = Get-ADOrganizationalUnit -Filter { Name -eq "Users" } `
                   -SearchBase $domainDN -SearchScope Subtree -ErrorAction SilentlyContinue |
               Select-Object -First 1

    if ($usersOU) {
        $SearchBase = $usersOU.DistinguishedName
        Write-Host "  Search Base : $SearchBase (auto-detected)" -ForegroundColor Gray
    }
    else {
        $SearchBase = $domainDN
        Write-Host "  WARNING: No 'Users' OU found — searching from domain root." -ForegroundColor Yellow
    }
}

Write-Host "  Max Threads : $MaxThreads" -ForegroundColor Gray
Write-Host "  Output File : $csvPath"    -ForegroundColor Gray
Write-Host ""

# ─────────────────────────────────────────────
# Step 1 — Retrieve enabled users + resolve
#           manager names IN THE MAIN THREAD
#           (AD module is reliable here)
# ─────────────────────────────────────────────
Write-Host "[Step 1/3] Querying Active Directory for enabled users..." -ForegroundColor Yellow

try {
    $adProperties = @(
        'GivenName','Surname','DisplayName','Description',
        'physicalDeliveryOfficeName','telephoneNumber','mail',
        'UserPrincipalName','mobile','Title',
        'Department','Company','Manager'
    )

    $rawUsers = @(
        Get-ADUser -Filter { Enabled -eq $true } `
                   -SearchBase  $SearchBase `
                   -SearchScope Subtree `
                   -Properties  $adProperties `
                   -ErrorAction Stop
    )

    if ($rawUsers.Count -eq 0) {
        Write-Host "WARNING: No enabled users found. Exiting." -ForegroundColor Yellow
        exit 0
    }

    Write-Host "  Found $($rawUsers.Count) enabled user(s)." -ForegroundColor Green
}
catch {
    Write-Host "ERROR: Failed to query Active Directory." -ForegroundColor Red
    Write-Host "  Detail: $_" -ForegroundColor Red
    exit 1
}

# Build a manager DN -> DisplayName cache to avoid redundant AD lookups
Write-Host "  Building manager name cache..." -ForegroundColor Gray
$managerCache = @{}
$managerDNs   = $rawUsers | Where-Object { $_.Manager } | Select-Object -ExpandProperty Manager -Unique

foreach ($dn in $managerDNs) {
    try {
        $mgr = Get-ADUser -Identity $dn -Properties DisplayName -ErrorAction Stop
        $managerCache[$dn] = if ($mgr.DisplayName) { $mgr.DisplayName } else { $mgr.SamAccountName }
    }
    catch {
        $managerCache[$dn] = "Lookup Failed"
    }
}
Write-Host "  Cached $($managerCache.Count) manager name(s)." -ForegroundColor Gray

# Pre-build plain hashtables — all property values extracted here in the
# main thread where the AD module is loaded and properties are accessible.
$userDataList = foreach ($u in $rawUsers) {
    $mgr = if ($u.Manager) { $managerCache[$u.Manager] } else { "" }

    @{
        FirstName   = if ($u.GivenName)                       { $u.GivenName.Trim() }                       else { "" }
        LastName    = if ($u.Surname)                         { $u.Surname.Trim() }                         else { "" }
        DisplayName = if ($u.DisplayName)                     { $u.DisplayName.Trim() }                     else { "" }
        Description = if ($u.Description)                     { $u.Description.Trim() }                     else { "" }
        Office      = if ($u.physicalDeliveryOfficeName)      { $u.physicalDeliveryOfficeName.Trim() }      else { "" }
        Telephone   = if ($u.telephoneNumber)                 { $u.telephoneNumber.Trim() }                 else { "" }
        Email       = if ($u.mail)                            { $u.mail.Trim() }                            else { "" }
        UPN         = if ($u.UserPrincipalName)               { $u.UserPrincipalName.Trim() }               else { "" }
        Mobile      = if ($u.mobile)                          { $u.mobile.Trim() }                          else { "" }
        Title       = if ($u.Title)                           { $u.Title.Trim() }                           else { "" }
        Department  = if ($u.Department)                      { $u.Department.Trim() }                      else { "" }
        Company     = if ($u.Company)                         { $u.Company.Trim() }                         else { "" }
        Manager     = if ($mgr)                               { $mgr }                                       else { "" }
    }
}

# ─────────────────────────────────────────────
# Step 2 — Multi-threaded object construction
# ─────────────────────────────────────────────
Write-Host ""
Write-Host "[Step 2/3] Building output records across $MaxThreads thread(s)..." -ForegroundColor Yellow

$syncHash         = [System.Collections.Hashtable]::Synchronized(@{})
$syncHash.Results = [System.Collections.Concurrent.ConcurrentBag[PSObject]]::new()
$syncHash.Errors  = [System.Collections.Concurrent.ConcurrentBag[string]]::new()

# Scriptblock only does PSCustomObject creation — no AD calls needed
$scriptBlock = {
    param($data, $syncHash)
    try {
        $syncHash.Results.Add([PSCustomObject]@{
            'First Name'      = $data.FirstName
            'Last Name'       = $data.LastName
            'Display Name'    = $data.DisplayName
            'Description'     = $data.Description
            'Office'          = $data.Office
            'Telephone'       = $data.Telephone
            'Email'           = $data.Email
            'User Logon Name' = $data.UPN
            'Mobile Number'   = $data.Mobile
            'Job Title'       = $data.Title
            'Department'      = $data.Department
            'Company'         = $data.Company
            'Manager'         = $data.Manager
        })
    }
    catch {
        $syncHash.Errors.Add("Error building record for '$($data.UPN)': $_")
    }
}

$runspacePool = [runspacefactory]::CreateRunspacePool(1, $MaxThreads)
$runspacePool.Open()

$jobs = foreach ($data in $userDataList) {
    $ps = [powershell]::Create()
    $ps.RunspacePool = $runspacePool
    $null = $ps.AddScript($scriptBlock).AddArgument($data).AddArgument($syncHash)
    [PSCustomObject]@{ PowerShell = $ps; Handle = $ps.BeginInvoke() }
}

$total = $jobs.Count
do {
    $done = @($jobs | Where-Object { $_.Handle.IsCompleted }).Count
    $pct  = [math]::Round(($done / $total) * 100)
    Write-Progress -Activity "Building output records" `
                   -Status   "$done of $total complete" `
                   -PercentComplete $pct
    Start-Sleep -Milliseconds 100
} while ($done -lt $total)

Write-Progress -Activity "Building output records" -Completed

foreach ($job in $jobs) {
    try   { $null = $job.PowerShell.EndInvoke($job.Handle) }
    catch { $syncHash.Errors.Add("EndInvoke error: $_") }
    $job.PowerShell.Dispose()
}
$runspacePool.Close()
$runspacePool.Dispose()

Write-Host "  Processed : $($syncHash.Results.Count) record(s)" -ForegroundColor Green

if ($syncHash.Errors.Count -gt 0) {
    Write-Host "  Warnings  : $($syncHash.Errors.Count) error(s):" -ForegroundColor Yellow
    $syncHash.Errors | ForEach-Object { Write-Host "    • $_" -ForegroundColor DarkYellow }
}

# ─────────────────────────────────────────────
# Step 3 — Export to CSV
# ─────────────────────────────────────────────
Write-Host ""
Write-Host "[Step 3/3] Exporting results to CSV..." -ForegroundColor Yellow

try {
    if ($syncHash.Results.Count -eq 0) {
        Write-Host "WARNING: No records to export." -ForegroundColor Yellow
        exit 0
    }

    $syncHash.Results |
        Sort-Object 'Display Name' |
        Export-Csv -Path $csvPath -NoTypeInformation -Encoding UTF8 -ErrorAction Stop

    Write-Host "  Status  : Export successful"           -ForegroundColor Green
    Write-Host "  File    : $csvPath"                    -ForegroundColor Green
    Write-Host "  Records : $($syncHash.Results.Count)"  -ForegroundColor Green
}
catch {
    Write-Host "ERROR: CSV export failed." -ForegroundColor Red
    Write-Host "  Detail: $_"              -ForegroundColor Red
    exit 1
}

Write-Host ""
Write-Host "Script completed successfully." -ForegroundColor Cyan
Write-Host ""
